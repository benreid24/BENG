243 bool Registry::addComponent(Entity entity, const T& component) {
244  const Component::IdType& id = Component::getId<T>();
245  std::unique_lock lock(entityMutex);
246 
247  // entity does not exist
248  if (entities.find(entity) == entities.end()) return false;
249 
250  // ensure pool exists
251  auto poolIter = componentPools.find(id);
252  if (poolIter == componentPools.end()) { poolIter = componentPools.try_emplace(id).first; }
253 
254  // ensure component link exists
255  auto linkIter = componentEntities.find(id);
256  if (linkIter == componentEntities.end()) { linkIter = componentEntities.try_emplace(id).first; }
257 
258  // ensure entity does not already have component
259  auto indexIter = entityComponentIterators.find(entity);
260  if (indexIter == entityComponentIterators.end()) {
261  indexIter = entityComponentIterators.try_emplace(entity).first;
262  }
263  if (indexIter->second.find(id) != indexIter->second.end()) return false;
264 
265  // add and track component
266  auto it = poolIter->second.add(component);
267  indexIter->second.emplace(id, it);
268  linkIter->second.insert(entity);
269 
270  invalidateViews(id);
271  return true;
272 }
