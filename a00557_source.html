144 Value str(SymbolTable& t, const std::vector<Value>& args) {
145  if (args.size() != 1) throw Error("str() takes 1 argument");
146  switch (args[0].getType()) {
147  case Value::TBool:
148  return Value(args[0].getAsBool() ? "true" : "false");
149  case Value::TNumeric: {
150  int v = args[0].getAsNum();
151  if (v == args[0].getAsNum()) return Value(std::to_string(v));
152  return Value(std::to_string(args[0].getAsNum()));
153  }
154  case Value::TString:
155  return Value(args[0].getAsString());
156  case Value::TArray: {
157  std::string result = "[";
158  if (!args[0].getAsArray().empty())
159  result += str(t, {*(args[0].getAsArray()[0])}).getAsString();
160  for (unsigned int i = 1; i < args[0].getAsArray().size(); ++i) {
161  result += ", " + str(t, {*(args[0].getAsArray()[i])}).getAsString();
162  }
163  result += "]";
164  return Value(result);
165  }
166  case Value::TRef: {
167  Value::CPtr r = args[0].getAsRef().lock();
168  if (!r) return Value("<expired reference>");
169  return str(t, {*r});
170  }
171  case Value::TFunction:
172  return Value("<function>");
173 
174  case Value::TVoid:
175  return Value("<void>");
176 
177  default:

179  }
